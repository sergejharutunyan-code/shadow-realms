   252→      towerCurrentFloor: 1,
   253→      towerAttempts: 5,
   254→      towerLastReset: Date.now(),
   255→      // Daily Dungeons
   256→      dungeonAttempts: {},
   257→      dungeonLastReset: Date.now(),
   258→      dungeonMaterials: {},
   259→      dungeonHighestStage: {},
   260→      heroesSummoned: 0,
   261→      gemsSpent: 0,
   262→      equipmentEquipped: 0,
   263→      campaignStagesWon: 0,
   264→      heroesLeveledUp: 0,
   265→      // Wandering Merchant
   266→      merchantWindowStart: 0,
   267→      merchantPurchases: {},
   268→      merchantLastSeen: 0,
   269→      // Daily Login Streak
   270→      loginStreak: 0,
   271→      loginStreakLastClaim: 0,
   272→      claimedStreakMilestones: [],
   273→      showStreakReward: false,
   274→      // Hero Skins — start with default skins owned
   275→      ownedSkins: HERO_SKINS.filter(s => s.cost === 0).map(s => s.id),
   276→      equippedSkins: {},
   277→
   278→      setScreen: (screen) => set({ screen }),
   279→
   280→      summon: (count, cost) => {
   281→        const state = get();
   282→        if (state.player.gems < cost) {
   283→          state.addNotification('Not enough gems! Visit the shop to get more!', 'warning');
   284→          return;
   285→        }
   286→        if (state.player.energy < 5) {
   287→          state.addNotification('Not enough energy! Wait for refill or buy from shop!', 'warning');
   288→          return;
   289→        }
   290→
   291→        const results: HeroInstance[] = [];
   292→        for (let i = 0; i < count; i++) {
   293→          const template = getSummonResult(state.player.vipLevel);
   294→          const instance = createHeroInstance(template);
   295→          results.push(instance);
   296→        }
   297→
   298→        // Check for duplicates and merge
   299→        const existingHeroes = [...state.heroes];
   300→        const newHeroes: HeroInstance[] = [];
   301→
   302→        for (const result of results) {
   303→          const existing = existingHeroes.find(h => h.templateId === result.templateId);
   304→          if (existing) {
   305→            // Merge - add experience to existing hero
   306→            existing.experience += 500;
   307→          } else {
   308→            newHeroes.push(result);
   309→          }
   310→        }
   311→
   312→        set({
   313→          player: { ...state.player, gems: state.player.gems - cost, energy: state.player.energy - 5 },
   314→          heroes: [...existingHeroes, ...newHeroes],
   315→          summonResults: results,
   316→          showSummonAnimation: true,
   317→          heroesSummoned: state.heroesSummoned + count,
   318→          gemsSpent: state.gemsSpent + cost,
   319→        });
   320→
   321→        // Update mission progress
   322→        get().updateMissionProgress('summon', count);
   323→
   324→        // Check for epic+ pulls
   325→        const bestRarity = results.reduce((best, h) => {
   326→          const order = ['common', 'uncommon', 'rare', 'epic', 'legendary', 'mythic'];
   327→          return order.indexOf(h.rarity) > order.indexOf(best) ? h.rarity : best;
   328→        }, 'common' as string);
   329→
   330→        if (bestRarity === 'mythic') {
   331→          state.addNotification('🌟 MYTHIC PULL! You summoned a MYTHIC champion! 🌟', 'epic');
   332→        } else if (bestRarity === 'legendary') {
   333→          state.addNotification('✨ LEGENDARY! An incredible champion joins your roster!', 'success');
   334→        } else if (bestRarity === 'epic') {
   335→          state.addNotification('💜 Epic champion acquired!', 'success');
   336→        }
   337→      },
   338→
   339→      addToTeam: (heroId) => {
   340→        const state = get();
   341→        if (state.team.length >= 5) {
   342→          state.addNotification('Team is full! Remove a hero first.', 'warning');
   343→          return;
   344→        }
   345→        if (state.team.includes(heroId)) return;
   346→        set({
   347→          team: [...state.team, heroId],
   348→          heroes: state.heroes.map(h => h.id === heroId ? { ...h, inTeam: true } : h),
   349→        });
   350→      },
   351→
   352→      removeFromTeam: (heroId) => {
   353→        const state = get();
   354→        if (state.team.length <= 1) {
   355→          state.addNotification('You need at least 1 hero in your team!', 'warning');
   356→          return;
   357→        }
   358→        set({
   359→          team: state.team.filter(id => id !== heroId),
   360→          heroes: state.heroes.map(h => h.id === heroId ? { ...h, inTeam: false } : h),
   361→        });
   362→      },
   363→
   364→      startBattle: (stage) => {
   365→        const state = get();
   366→        const stageData = CAMPAIGN_STAGES[stage - 1];
   367→        if (!stageData) return;
   368→        if (state.player.energy < stageData.energyCost) {
   369→          state.addNotification('Not enough energy!', 'warning');
   370→          return;
   371→        }
   372→
   373→        const teamHeroes = state.team
   374→          .map(id => state.heroes.find(h => h.id === id))
   375→          .filter((h): h is HeroInstance => h !== undefined);
   376→
   377→        if (teamHeroes.length === 0) {
   378→          state.addNotification('Add heroes to your team first!', 'warning');
   379→          return;
   380→        }
   381→
   382→        const playerTeam = teamHeroes.map(createBattleHero);
   383→        const enemyTeam = generateEnemyTeam(stage, stageData.enemyFaction);
   384→
   385→        set({
   386→          player: { ...state.player, energy: state.player.energy - stageData.energyCost },
   387→          battle: {
   388→            isActive: true,
   389→            playerTeam,
   390→            enemyTeam,
   391→            currentTurn: 'player',
   392→            turnIndex: 0,
   393→            turnNumber: 1,
   394→            log: ['⚔️ Battle begins! Turn 1'],
   395→            result: null,
   396→            rewards: null,
   397→            autoPlay: false,
   398→            speed: 1,
   399→          },
   400→          screen: 'battle',
   401→        });
   402→
   403→        // Auto-start battle simulation
   404→        get().executeBattleTurn();
   405→      },
   406→
   407→      executeBattleTurn: () => {
   408→        const state = get();
   409→        if (!state.battle || state.battle.result) return;
   410→
   411→        const battle = { ...state.battle };
   412→        const log = [...battle.log];
   413→        const playerTeam = battle.playerTeam.map(h => ({ ...h }));
   414→        const enemyTeam = battle.enemyTeam.map(h => ({ ...h }));
   415→
   416→        const livingPlayers = playerTeam.filter(h => h.isAlive);
   417→        const livingEnemies = enemyTeam.filter(h => h.isAlive);
   418→
   419→        if (livingPlayers.length === 0) {
   420→          set({
   421→            battle: {
   422→              ...battle,
   423→              playerTeam,
   424→              enemyTeam,
   425→              log: [...log, '💀 DEFEAT! Your team has been destroyed.'],
   426→              result: 'defeat',
   427→              rewards: null,
   428→            },
   429→          });
   430→          return;
   431→        }
   432→        if (livingEnemies.length === 0) {
   433→          const stageData = CAMPAIGN_STAGES[state.player.campaignStage - 1];
   434→          const goldReward = stageData?.rewards.gold || 100;
   435→          const expReward = stageData?.rewards.experience || 50;
   436→          const gemReward = stageData?.rewards.gems || 5;
   437→
   438→          // Apply VIP bonuses
   439→          const vipGoldBonus = 1 + state.player.vipLevel * 0.1;
   440→          const actualGold = Math.floor(goldReward * vipGoldBonus);
   441→
   442→          // Level up player
   443→          let newExp = state.player.experience + expReward;
   444→          let newLevel = state.player.level;
   445→          let newGold = state.player.gold + actualGold;
   446→          let newGems = state.player.gems + gemReward;
   447→
   448→          while (newExp >= getExperienceForLevel(newLevel)) {
   449→            newExp -= getExperienceForLevel(newLevel);
   450→            newLevel++;
   451→            newGems += 10;
   452→          }
   453→
   454→          // Level up heroes
   455→          const updatedHeroes = state.heroes.map(h => {
   456→            if (state.team.includes(h.id)) {
   457→              let heroExp = h.experience + expReward;
   458→              let heroLevel = h.level;
   459→              while (heroExp >= getExperienceForLevel(heroLevel)) {
   460→                heroExp -= getExperienceForLevel(heroLevel);
   461→                heroLevel++;
   462→              }
   463→              const stats = getHeroStats(
   464→                { baseAttack: h.attack / (1 + (h.level - 1) * 0.08) / (1 + (h.stars - 1) * 0.15), baseDefense: h.defense / (1 + (h.level - 1) * 0.08) / (1 + (h.stars - 1) * 0.15), baseHealth: h.health / (1 + (h.level - 1) * 0.08) / (1 + (h.stars - 1) * 0.15), baseSpeed: h.speed / (1 + (h.stars - 1) * 0.05) } as HeroTemplate,
   465→                heroLevel,
   466→                h.stars
   467→              );
   468→              return { ...h, level: heroLevel, experience: heroExp, attack: stats.attack, defense: stats.defense, health: stats.health, speed: stats.speed };
   469→            }
   470→            return h;
   471→          });
   472→
   473→          const newCampaignStage = Math.max(state.player.campaignStage, stageData ? stageData.id + 1 : state.player.campaignStage);
   474→
   475→          // Equipment drop chance (15% base + 2% per stage)
   476→          const dropChance = 0.15 + (stageData?.id || 1) * 0.02;
   477→          const droppedEquipment = Math.random() < dropChance ? get().generateRandomEquipment(stageData ? stageData.id / 50 : 0) : null;
   478→          const newEquipment = droppedEquipment ? [...state.equipment, droppedEquipment] : state.equipment;
   479→
   480→          // Get dropped equipment name for notification
   481→          let dropName = '';
   482→          if (droppedEquipment) {
   483→            const dropTemplate = EQUIPMENT_TEMPLATES.find(t => t.id === droppedEquipment.templateId);
   484→            dropName = dropTemplate ? `${dropTemplate.icon} ${dropTemplate.name}` : 'Equipment';
   485→          }
   486→
   487→          set({
   488→            player: {
   489→              ...state.player,
   490→              level: newLevel,
   491→              experience: newExp,
   492→              gold: newGold,
   493→              gems: newGems,
   494→              campaignStage: newCampaignStage,
   495→            },
   496→            heroes: updatedHeroes,
   497→            equipment: newEquipment,
   498→            battlesWon: state.battlesWon + 1,
   499→            totalBattles: state.totalBattles + 1,
   500→            battle: {
   501→              ...battle,
   502→              playerTeam,
   503→              enemyTeam,
   504→              log: [...log, `🏆 VICTORY! +${actualGold} Gold, +${expReward} XP, +${gemReward} Gems`, droppedEquipment ? `🎁 Equipment Drop: ${dropName}!` : ''].filter(Boolean),
   505→              result: 'victory',
   506→              rewards: { gold: actualGold, experience: expReward, gems: gemReward },
   507→            },
   508→          });
   509→
   510→          if (droppedEquipment) {
   511→            setTimeout(() => get().addNotification(`🎁 Equipment Drop: ${dropName}!`, 'epic'), 500);
   512→          }
   513→          // Check achievements after victory
   514→          setTimeout(() => get().checkAchievements(), 100);
   515→          return;
   516→        }
   517→
   518→        // Player turn
   519→        for (const hero of livingPlayers) {
   520→          const target = livingEnemies[Math.floor(Math.random() * livingEnemies.length)];
   521→          if (!target) continue;
   522→
   523→          const isCrit = Math.random() < hero.critRate;
   524→          let damage = Math.floor(hero.attack * (1.5 + Math.random() * 0.5) * (isCrit ? hero.critDamage : 1));
   525→          const defReduction = target.defense / (target.defense + 500);
   526→          damage = Math.floor(damage * (1 - defReduction));
   527→
   528→          // Track max damage for achievements
   529→          if (damage > state.maxDamage) {
   530→            set({ maxDamage: damage });
   531→          }
   532→
   533→          const enemyIdx = enemyTeam.findIndex(e => e.id === target.id);
   534→          enemyTeam[enemyIdx] = {
   535→            ...target,
   536→            currentHealth: Math.max(0, target.currentHealth - damage),
   537→            isAlive: target.currentHealth - damage > 0,
   538→          };
